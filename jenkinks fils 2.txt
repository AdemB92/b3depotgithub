pipeline {
    agent any
    stages {
        stage('clear workspace') {
            steps {
                cleanWs()
            }
        }

        stage('install docker.io') {
            steps {
                sh 'sudo apt autoremove docker.io docker-compose -y'
                sh 'sudo apt update'
                sh 'sudo apt install apt-transport-https ca-certificates curl software-properties-common -y'
                sh 'sudo apt install docker.io -y'
                sh 'sudo apt update'
                sh 'sudo chmod 666 /var/run/docker.sock'
                sh 'sudo usermod -aG docker $USER'
                sh 'sudo systemctl status docker.service'
                sh 'sudo systemctl start docker.service'
            }
        }

        stage('install docker') {
            steps {
                sh 'sudo apt update'
                sh 'sudo apt upgrade -y'
                sh 'sudo apt install ca-certificates curl gnupg lsb-release -y'

                sh 'sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg -o /tmp/docker.gpg'
                sh 'sudo gpg --batch --yes --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg /tmp/docker.gpg'

                sh "echo \"deb [arch=\$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \$(lsb_release -cs) stable\" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null"

                sh 'sudo apt update'
                sh 'sudo apt install docker-ce docker-ce-cli containerd.io -y'

                sh 'sudo chmod 666 /var/run/docker.sock'
                sh 'sudo usermod -aG docker $USER'

                sh 'sudo docker run hello-world'
            }
        }

        stage('Test create container') {
            steps {
                sh 'docker image ls'
                sh 'sudo docker run -d --name conteneur01 --hostname conteneur01 ubuntu/apache2'
                sh 'docker ps'
            }
        }
    }
}
